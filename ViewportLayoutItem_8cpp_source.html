112void ViewportLayoutItem::_showAssignmentMenu() {
113 const int slotCount = static_cast<int>(m_def.panes.size());
114
115 // Build current slot→plane mapping from the definition.
116 std::vector<int> current(slotCount);
117 for (const auto& pane : m_def.panes) {
118 if (pane.slotIndex >= 0 && pane.slotIndex < slotCount)
119 current[pane.slotIndex] = pane.planeIndex;
120 }
121
122 QMenu menu(this);
123 menu.setTitle(tr("Assign Planes"));
124
125 std::vector<int> assignment = current;
126
127 struct SlotMenu {
128 QMenu* menu;
129 QAction* actions[3];
130 };
131 std::vector<SlotMenu> slotMenus(slotCount);
132
133 for (int s = 0; s < slotCount; ++s) {
134 SlotMenu& sm = slotMenus[s];
135 sm.menu = menu.addMenu(tr("Slot %1 ").arg(s + 1));
136
137 for (int p = 0; p < 3; ++p) {
138 sm.actions[p] = sm.menu->addAction(tr(kPlaneNames[p]));
139 sm.actions[p]->setCheckable(true);
140 sm.actions[p]->setChecked(assignment[s] == p);
141 }
142 }
143
144 // Wire up actions: selecting plane p for slot s swaps with the slot that
145 // already holds p, so every plane appears in exactly one slot.
146 for (int s = 0; s < slotCount; ++s) {
147 for (int p = 0; p < 3; ++p) {
148 connect(slotMenus[s].actions[p], &QAction::triggered, this,
149 [this, s, p, slotCount, slotMenus, assignment]() mutable {
150 // Find which slot currently holds plane p and swap.
151 for (int other = 0; other < slotCount; ++other) {
152 if (other != s && assignment[other] == p) {
153 assignment[other] = assignment[s]; // give s's old plane to other
154 break;
155 }
156 }
157 assignment[s] = p;
158 emit slotAssignmentRequested(assignment);
159 });
160 }
161 }
162
163 menu.exec(QCursor::pos());
164}
