29void OverlayLayoutAdapter::Sync(const ui::ViewportLayoutDefinition& def) {
30 // Rebuild plane→pane map and cache pane data.
31 m_currentPane.fill(nullptr);
32 for (const auto& pane : def.panes) {
33 const int idx = pane.planeIndex;
34 if (idx >= 0 && idx < kNumPlanes && !m_currentPane[idx]) {
35 m_paneStorage[idx] = pane;
36 m_currentPane[idx] = &m_paneStorage[idx];
37 }
38 }
39
40 for (int i = 0; i < kNumPlanes; ++i)
41 m_layoutVisible[i] = (m_currentPane[i] != nullptr);
42
43 // Apply to orientation overlays.
44 for (int i = 0; i < static_cast<int>(m_orientationOverlays.size()); ++i) {
45 _ApplyVisibility(m_orientationOverlays[i],
46 i < kNumPlanes ? m_orientationUserEnabled[i] : false,
47 i < kNumPlanes ? m_layoutVisible[i] : false,
48 i < kNumPlanes ? m_currentPane[i] : nullptr);
49 }
50
51 // Apply to corner annotation overlays.
52 for (int i = 0; i < static_cast<int>(m_cornerAnnotationOverlays.size()); ++i) {

54 i < kNumPlanes ? m_cornerUserEnabled[i] : false,
