107 // ── VTK callbacks ─────────────────────────────────────────────────────
108 static void OnRenderStart(vtkObject*, unsigned long, void* clientData, void*) {
109 auto* self = static_cast<Impl*>(clientData);
110 self->m_renderStart = std::chrono::steady_clock::now();
111 }
112
113 static void OnRenderEnd(vtkObject*, unsigned long, void* clientData, void*) {
114 auto* self = static_cast<Impl*>(clientData);
115 const auto renderEnd = std::chrono::steady_clock::now();
116 const double renderMs = std::chrono::duration<double, std::milli>(
117 renderEnd - self->m_renderStart)
118 .count();
119
120 for (auto& slot : self->m_slots) {
121 const auto now = renderEnd;
122 double frameMs = 0.0;
123 if (!slot.firstFrame) {
124 frameMs = std::chrono::duration<double, std::milli>(
125 now - slot.lastFrameEnd)
126 .count();
127 }
128 slot.firstFrame = false;
129 slot.lastFrameEnd = now;
130
131 slot.stats.frameTimeMs = frameMs;
132 slot.stats.renderTimeMs = renderMs;
133 slot.stats.frameCount++;
134
135 if (self->m_statsOverlayEnabled && slot.overlay)
136 slot.overlay->Update(slot.config.label, slot.stats);
137
138 self->NotifyFrameRendered(slot.index, slot.stats);
139 }
140 }

162int ViewportManager::AddViewport(const ViewportConfig& config) {
163 m_impl->EnsureRenderWindow();
164
165 detail::ViewportSlot slot;
166 slot.index = m_impl->m_nextIndex++;
167 slot.config = config;
168 slot.renderer = vtkSmartPointer<vtkRenderer>::New();
169
170 m_impl->ApplyViewportToRenderer(slot);
171 m_impl->m_renderWindow->AddRenderer(slot.renderer);
172
173 // Keep layer count up to date
174 const int maxLayer = slot.config.layer + 1;
175 if (m_impl->m_renderWindow->GetNumberOfLayers() < maxLayer)
176 m_impl->m_renderWindow->SetNumberOfLayers(maxLayer);
177
178 slot.overlay =
179 std::make_unique<RenderStatsOverlay>(slot.renderer);
180 slot.overlay->SetVisible(m_impl->m_statsOverlayEnabled);
181
182 const int idx = slot.index;
183 m_impl->m_slots.push_back(std::move(slot));
184
185 for (auto* obs : m_impl->m_observers)
186 obs->OnViewportAdded(idx);
187
188 return idx;
189}
