153void OrientationMarkerOverlay::_DrawLabel(QPainter& p, const QFontMetrics& fm,
154 const QString& text, Qt::Alignment align,
155 int vpLeft, int vpTop, int vpRight, int vpBot,
156 int vpW, int vpH) const {
157 if (text.isEmpty())
158 return;
159
160 const int textW = fm.horizontalAdvance(text);
161 const int textH = fm.height();
162
163 int x = 0, y = 0;
164 if (align & Qt::AlignLeft) {
165 x = vpLeft + m_margin;
166 y = vpTop + (vpH - textH) / 2;
167 } else if (align & Qt::AlignRight) {
168 x = vpRight - textW - m_margin;
169 y = vpTop + (vpH - textH) / 2;
170 } else if (align & Qt::AlignTop) {
171 x = vpLeft + (vpW - textW) / 2;
172 y = vpTop + m_margin;
173 } else if (align & Qt::AlignBottom) {
174 x = vpLeft + (vpW - textW) / 2;
175 y = vpBot - textH - m_margin;
176 } else {
177 return;
178 }
179
180 p.setPen(Qt::black);
181 p.drawText(x + 1, y + 1 + fm.ascent(), text);
182 p.setPen(m_textColor);
183 p.drawText(x, y + fm.ascent(), text);
184}
