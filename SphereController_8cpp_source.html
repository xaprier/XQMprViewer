70void SphereController::AddInteractor(vtkSmartPointer<vtkRenderWindowInteractor> interactor) {
71 if (!interactor)

88void SphereController::Cleanup() {
89 for (const auto& e : m_rendererEntries)
90 if (e.renderer)
91 e.renderer->RemoveActor(e.actor);
92
93 for (auto& interactor : m_interactors) {
94 if (!interactor)
95 continue;
96 interactor->RemoveObserver(m_leftDownCmd);
97 interactor->RemoveObserver(m_mouseMoveCmd);
98 interactor->RemoveObserver(m_leftUpCmd);
99 }
100
101 m_rendererEntries.clear();
102 m_interactors.clear();
103 m_isDragging = false;
104 m_activeRenderer = nullptr;

183void SphereController::RenderAll() {
184 if (m_scheduler) {
185 m_scheduler->RequestRenderAll();
186 m_scheduler->Flush();
187 return;
188 }
189
190 // Fallback when no scheduler is wired (should not happen in normal usage).
191 std::vector<vtkRenderWindow*> seen;
192 for (const auto& e : m_rendererEntries) {
193 if (!e.renderer)
194 continue;
195 auto* win = e.renderer->GetRenderWindow();
196 if (!win)
197 continue;
198 if (std::find(seen.begin(), seen.end(), win) == seen.end()) {
199 seen.push_back(win);
200 win->Render();
201 }
202 }
203}

207void SphereController::OnLeftButtonDown(vtkObject* caller, unsigned long, void* clientData, void*) {
208 auto* self = static_cast<SphereController*>(clientData);
209
210 self->m_leftDownCmd->SetAbortFlag(0); // important
211
212 auto* interactor = vtkRenderWindowInteractor::SafeDownCast(caller);
213 if (!interactor)
214 return;
215
216 int pos[2];
217 interactor->GetEventPosition(pos);
218
219 // FindPokedRenderer works for BOTH modes:
220 // viewport mode (1 window, N renderers): returns renderer at mouse pos
221 // multiwindow (N windows, 1 renderer each): returns the single renderer
222 vtkRenderer* renderer = interactor->FindPokedRenderer(pos[0], pos[1]);
223 if (!renderer)
224 return;
225
226 self->m_picker->Pick(pos[0], pos[1], 0.0, renderer);
227
228 auto* expectedActor = self->ActorFor(renderer);
229 if (self->m_picker->GetActor() != expectedActor)
230 return;
231
232 self->m_isDragging = true;
233 self->m_activeRenderer = renderer;
234 self->m_activePlane = self->PlaneFor(renderer);
235
236 // Prevent the interactor style from entering window/level (or any other)
237 // interaction state. Without this, the style owns the drag and its
238 // OnLeftButtonUp can race with ours, leaving m_isDragging stuck true.
239 // NOTE: SetEnabled(0) must NOT be used — it fires DisableEvent which
240 // vtkResliceCursorWidget listens to, hiding the DICOM image.
241 self->m_leftDownCmd->SetAbortFlag(1);
242}

244void SphereController::OnMouseMove(vtkObject* caller, unsigned long, void* clientData, void*) {
245 auto* self = static_cast<SphereController*>(clientData);
246 if (!self->m_isDragging || !self->m_activeRenderer)
247 return;
248
249 auto* interactor = vtkRenderWindowInteractor::SafeDownCast(caller);
250 if (!interactor)
251 return;
252
253 int pos[2];
254 interactor->GetEventPosition(pos);
255
256 // Build world-space ray from screen position
257 self->m_activeRenderer->SetDisplayPoint(pos[0], pos[1], 0.0);
258 self->m_activeRenderer->DisplayToWorld();
259 double near4[4];
260 self->m_activeRenderer->GetWorldPoint(near4);
261
262 self->m_activeRenderer->SetDisplayPoint(pos[0], pos[1], 1.0);
263 self->m_activeRenderer->DisplayToWorld();
264 double far4[4];
265 self->m_activeRenderer->GetWorldPoint(far4);
266
267 if (near4[3] == 0.0 || far4[3] == 0.0)
268 return;
269
270 const double rayNear[3] = {near4[0] / near4[3], near4[1] / near4[3], near4[2] / near4[3]};
271 const double rayFar[3] = {far4[0] / far4[3], far4[1] / far4[3], far4[2] / far4[3]};
272
273 // Drag plane passes through sphere center; normal is the view axis of the active viewport
274 double center[3];
275 self->m_sphereSource->GetCenter(center);
276
277 double normal[3] = {0.0, 0.0, 0.0};
278 switch (self->m_activePlane) {
279 case DragPlane::Axial:
280 normal[2] = 1.0;
281 break;
282 case DragPlane::Coronal:
283 normal[1] = 1.0;
284 break;
285 case DragPlane::Sagittal:
286 normal[0] = 1.0;
287 break;
288 }
289
290 double t = 0.0;
291 double intersect[3] = {0.0, 0.0, 0.0};
292 if (!vtkPlane::IntersectWithLine(rayNear, rayFar, normal, center, t, intersect))
293 return;
294
295 self->SetPosition({intersect[0], intersect[1], intersect[2]});
296 // Abort so the style doesn't do window/level simultaneously with sphere drag.
297 // SetPosition emits SphereMoved → SliceController::OnSphereUpdated → RenderAll,
298 // so the scheduler already has a pending flush; no extra render call needed here.
299 // self->m_mouseMoveCmd->SetAbortFlag(1);
300}
