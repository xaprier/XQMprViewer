82void FPSOverlay::_OnFrameSwapped() {
83 const qint64 nowNs = m_frameTimer.nsecsElapsed();
84
85 if (m_lastFrameNs > 0) {
86 m_frameTimestamps.push_back(nowNs);
87
88 // Drop timestamps older than the averaging window.
89 const qint64 cutoff = nowNs - m_windowNs;
90 while (!m_frameTimestamps.empty() && m_frameTimestamps.front() < cutoff)
91 m_frameTimestamps.pop_front();
92
93 if (m_frameTimestamps.size() >= 2) {
94 const qint64 spanNs = m_frameTimestamps.back() - m_frameTimestamps.front();
95 const int frames = static_cast<int>(m_frameTimestamps.size()) - 1;
96 const double avgFrameMs = static_cast<double>(spanNs) / frames / 1'000'000.0;
97 const double avgFps = frames * 1'000'000'000.0 / static_cast<double>(spanNs);
98 _UpdateText(avgFrameMs, avgFps);
99 }
100 }
101
102 m_lastFrameNs = nowNs;
103}
