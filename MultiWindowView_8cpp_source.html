110void MultiWindowView::_RebuildGrid(const ViewportLayoutDefinition& def) {
111 // Remove all widgets from the grid without deleting them.
112 for (int i = 0; i < 3; ++i) {
113 if (m_vtkWidgets[i]) {
114 m_grid->removeWidget(m_vtkWidgets[i]);
115 m_vtkWidgets[i]->hide();
116 }
117 }
118
119 // Clear stretch factors.
120 for (int r = 0; r < m_grid->rowCount(); ++r)
121 m_grid->setRowStretch(r, 0);
122 for (int c = 0; c < m_grid->columnCount(); ++c)
123 m_grid->setColumnStretch(c, 0);
124
125 const ViewportLayoutType type = def.type;
126
127 switch (type) {
128 case ViewportLayoutType::SingleAxial:
129 case ViewportLayoutType::SingleCoronal:
130 case ViewportLayoutType::SingleSagittal: {
131 // One full-screen widget determined by the pane's planeIndex.
132 int plane = def.panes.empty() ? 0 : def.panes[0].planeIndex;
133 if (plane < 0 || plane > 2) plane = 0;
134 m_grid->addWidget(m_vtkWidgets[plane], 0, 0, 1, 1);
135 m_vtkWidgets[plane]->show();
136 m_grid->setRowStretch(0, 1);
137 m_grid->setColumnStretch(0, 1);
138 break;
139 }
140
141 case ViewportLayoutType::HorizontalSplit: {
142 // Three equal columns. The pane order from the definition gives us
143 // the mapping: slot 0 → col 0, slot 1 → col 1, slot 2 → col 2.
144 for (const auto& pane : def.panes) {
145 const int plane = pane.planeIndex;
146 const int col = pane.slotIndex;
147 if (plane < 0 || plane > 2 || col < 0 || col > 2) continue;
148 m_grid->addWidget(m_vtkWidgets[plane], 0, col, 1, 1);
149 m_vtkWidgets[plane]->show();
150 m_grid->setColumnStretch(col, 1);
151 }
152 m_grid->setRowStretch(0, 1);
153 break;
154 }
155
156 case ViewportLayoutType::VerticalSplit: {
157 // Three equal rows. slot 0 → row 2 (bottom), slot 1 → row 1, slot 2 → row 0 (top).
158 // VTK y=0 is bottom; visually we want slot 0 at bottom → Qt row 2.
159 for (const auto& pane : def.panes) {
160 const int plane = pane.planeIndex;
161 const int qtRow = 2 - pane.slotIndex; // invert: slot 0 = bottom row
162 if (plane < 0 || plane > 2 || qtRow < 0 || qtRow > 2) continue;
163 m_grid->addWidget(m_vtkWidgets[plane], qtRow, 0, 1, 1);
164 m_vtkWidgets[plane]->show();
165 m_grid->setRowStretch(qtRow, 1);
166 }
167 m_grid->setColumnStretch(0, 1);
168 break;
169 }
170
171 case ViewportLayoutType::TwoPlusOne: {
172 // slot 0: left full-height (col 0, rows 0-1)
173 // slot 1: right top (col 1, row 0)
174 // slot 2: right bottom (col 1, row 1)
175 for (const auto& pane : def.panes) {
176 const int plane = pane.planeIndex;
177 if (plane < 0 || plane > 2) continue;
178 if (pane.slotIndex == 0) {
179 m_grid->addWidget(m_vtkWidgets[plane], 0, 0, 2, 1);
180 } else if (pane.slotIndex == 1) {
181 m_grid->addWidget(m_vtkWidgets[plane], 0, 1, 1, 1);
182 } else if (pane.slotIndex == 2) {
183 m_grid->addWidget(m_vtkWidgets[plane], 1, 1, 1, 1);
184 }
185 m_vtkWidgets[plane]->show();
186 }
187 m_grid->setColumnStretch(0, 1);
188 m_grid->setColumnStretch(1, 1);
189 m_grid->setRowStretch(0, 1);
190 m_grid->setRowStretch(1, 1);
191 break;
192 }
193
194 case ViewportLayoutType::TwoPlusOneReversed: {
195 // slot 0: top-left (row 0, col 0)
196 // slot 1: top-right (row 0, col 1)
197 // slot 2: bottom full-width(row 1, cols 0-1)
198 for (const auto& pane : def.panes) {
199 const int plane = pane.planeIndex;
200 if (plane < 0 || plane > 2) continue;
201 if (pane.slotIndex == 0) {
202 m_grid->addWidget(m_vtkWidgets[plane], 0, 0, 1, 1);
203 } else if (pane.slotIndex == 1) {
204 m_grid->addWidget(m_vtkWidgets[plane], 0, 1, 1, 1);
205 } else if (pane.slotIndex == 2) {
206 m_grid->addWidget(m_vtkWidgets[plane], 1, 0, 1, 2);
207 }
208 m_vtkWidgets[plane]->show();
209 }
210 m_grid->setColumnStretch(0, 1);
211 m_grid->setColumnStretch(1, 1);
212 m_grid->setRowStretch(0, 1);
213 m_grid->setRowStretch(1, 1);
214 break;
215 }
216 }
