160void CornerAnnotationOverlay::paintEvent(QPaintEvent* /*event*/) {
161 if (!m_enabled)
162 return;
163
164 const QString text = _BuildText();
165 if (text.isEmpty())
166 return;
167
168 if (m_fontDirty) {
169 m_cachedFont = font();
170 m_cachedFont.setPointSize(m_fontSize);
171 m_cachedFont.setBold(true);
172 m_cachedMetrics = QFontMetrics(m_cachedFont);
173 m_fontDirty = false;
174 }
175
176 const QStringList lines = text.split('\n');
177 const QFontMetrics& fm = m_cachedMetrics;
178 const int lineH = fm.height();
179 int maxW = 0;
180 for (const auto& line : lines)
181 maxW = std::max(maxW, fm.horizontalAdvance(line));
182 const int blockH = lines.size() * lineH;
183
184 // Viewport sub-region
185 const int totalW = width();
186 const int totalH = height();
187 const int vpLeft = static_cast<int>(m_vpXMin * totalW);
188 const int vpTop = static_cast<int>(m_vpYMin * totalH);
189 const int vpRight = static_cast<int>(m_vpXMax * totalW);
190 const int vpBot = static_cast<int>(m_vpYMax * totalH);
191 const int vpW = vpRight - vpLeft;
192 const int vpH = vpBot - vpTop;
193 if (vpW <= 0 || vpH <= 0)
194 return;
195
196 // Block origin from position
197 int bx = 0, by = 0;
198 switch (m_position) {
199 case OverlayPosition::TopLeft:
200 bx = vpLeft + m_margin;
201 by = vpTop + m_margin;
202 break;
203 case OverlayPosition::TopRight:
204 bx = vpRight - maxW - m_margin;
205 by = vpTop + m_margin;
206 break;
207 case OverlayPosition::TopCenter:
208 bx = vpLeft + (vpW - maxW) / 2;
209 by = vpTop + m_margin;
210 break;
211 case OverlayPosition::BottomLeft:
212 bx = vpLeft + m_margin;
213 by = vpBot - blockH - m_margin;
214 break;
215 case OverlayPosition::BottomRight:
216 bx = vpRight - maxW - m_margin;
217 by = vpBot - blockH - m_margin;
218 break;
219 case OverlayPosition::BottomCenter:
220 bx = vpLeft + (vpW - maxW) / 2;
221 by = vpBot - blockH - m_margin;
222 break;
223 }
224
225 QPainter p(this);
226 p.setRenderHint(QPainter::TextAntialiasing);
227 p.setClipRect(vpLeft, vpTop, vpW, vpH);
228 p.setFont(m_cachedFont);
229
230 // Semi-transparent background box — same style as FPSOverlay.
231 const QRect bgRect(bx - m_padding,
232 by - m_padding,
233 maxW + 2 * m_padding,
234 blockH + 2 * m_padding);
235 p.setPen(Qt::NoPen);
236 p.setBrush(QColor(0, 0, 0, 120));
237 p.drawRoundedRect(bgRect, 3, 3);
238
239 for (int li = 0; li < lines.size(); ++li) {
240 const int x = bx;
241 const int y = by + li * lineH + fm.ascent();
242 p.setPen(Qt::black);
243 p.drawText(x + 1, y + 1, lines[li]);
244 p.setPen(m_textColor);
245 p.drawText(x, y, lines[li]);
246 }
247}

251QString CornerAnnotationOverlay::_BuildText() const {
252 QStringList lines;
253
254 if (m_showViewName && !m_viewName.isEmpty())
255 lines << m_viewName;
256
257 if (m_viewer) {
258 if (m_showSliceInfo) {
259 const int cur = m_viewer->GetSlice();
260 const int minS = m_viewer->GetSliceMin();
261 const int maxS = m_viewer->GetSliceMax();
262 const int count = maxS - minS + 1;
263 lines << tr("Slice: %1 / %2").arg(cur - minS + 1).arg(count);
264 }
265
266 if (m_showWindowLevel) {
267 // Read from ImageActor's property — these are the values actually
268 // applied to the rendered image, set by SliceController::SetupPipeline().
269 if (auto* actor = m_viewer->GetImageActor()) {
270 if (auto* prop = actor->GetProperty()) {
271 const double ww = prop->GetColorWindow();
272 const double wl = prop->GetColorLevel();
273 lines << tr("W: %1 L: %2")
274 .arg(ww, 0, 'f', 0)
275 .arg(wl, 0, 'f', 0);
276 }
277 }
278 }
279
280 if (m_showSpacing) {
281 if (auto* img = m_viewer->GetInput()) {
282 double sp[3];
283 img->GetSpacing(sp);
284
285 // Show only the through-plane (slice-normal) spacing for each
286 // orientation — the axis perpendicular to the viewed plane.
287 // Orientation enum: YZ=0 (Sagittal,X=sp[0]),
288 // XZ=1 (Coronal, Y=sp[1]),
289 // XY=2 (Axial, Z=sp[2]).
290 const int orientation = m_viewer->GetSliceOrientation();
291 static const char* kAxisLabel[3] = {"X", "Y", "Z"};
292 const double throughPlaneSpacing = sp[orientation];
293 const char* axisLabel = kAxisLabel[orientation];
294
295 lines << tr("Sp(%1): %2 mm")
296 .arg(QLatin1String(axisLabel))
297 .arg(throughPlaneSpacing, 0, 'f', 2);
298 }
299 }
300 }
301
302 return lines.join('\n');
303}
